`WIN ~` -> Launch Kitty
`WIN E` -> Launch Files
`WIN Z` -> Launch Browser
`WIN Space` -> Open Launcher
`WIN =` -> Launch Qalc
`CTRL SHIFT Esc` -> Launch Btop

`WIN L` -> Lock
`CTRL ALT DELETE` -> Logout

`WIN 0-9` -> Go to workpace
`WIN SHIFT 0-9` -> Move to workspace
